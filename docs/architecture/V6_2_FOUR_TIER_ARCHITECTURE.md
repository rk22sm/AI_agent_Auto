        this.learningMatrix.set(pattern.type, [])
      }

      existingPatterns.push({
        pattern: pattern,
        timestamp: Date.now(),
        effectiveness: pattern.effectiveness
      })

      // Keep only recent patterns
      if (existingPatterns.length > 100) {
        existingPatterns.shift()
      }
    }

function selectAgentsForTier(tier, taskComplexity, taskType) {
  const availableAgents = getTierAgents(tier)

  // Score each agent based on task fit

  // Sort by score and select top agents
  availableAgents.sort((a, b) => b.score - a.score)

    case 'simple':

    case 'moderate':
      selectedAgents.push(availableAgents[0], availableAgents[1]) // 2 agents

      break
    case 'critical':
      selectedAgents.push(...availableAgents.slice(0, 4)) // 4 agents
      break
  }

  return selectedAgents
}
